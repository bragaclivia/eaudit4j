33  package org.beiter.michael.eaudit4j.common;
34  
35  import java.util.List;
36  
37  /**
38   * This class implements an interface to represent audit events, an event being comprised of a set of {@link Field}s.
39   */
40  public interface Event {
41  
42      /**
43       * Each implementation of the {@code Event} interface comes with a JSON serialization.
44       * <p>
45       * This method returns the record format provided in the implementation's JSON serializer. See the library's
46       * This method returns the record format provided in the implementation's JSON serializer. See the library's
47       * documentation on specifications of supported record format versions.
48       *
49       * @return The record format version implemented by the {@code Event} implementation's JSON serializer
50       */
51      String getRecordFormatVersion();
52  
53      /**
54       * Add a field to the event.
55       * <p>
56       * Note that implementations of this method <strong>should</strong> create a defensive copy of the provided
57       * {@code Field}), and stores the copy in the event. You need to explicitly clear the data provided to this method
58       * if it contains confidential information that you wish to destroy.
59       *
60       * @param field The field to add
61       * @return The name of the field that was added to the event
62       */
63      String setField(Field field);
64  
65      /**
66       * Checks if the event contains a field with the specified name.
67       *
68       * @param fieldName The field name to check for
69       * @return {@code true} if the field is registered with the event, {@code false} otherwise
70       */
71      boolean containsField(String fieldName);
72  
73      /**
74       * Remove a field from the event.
75       *
76       * @param fieldName The name of the field to remove
77       * @return {@code true} if the field could be removed, {@code false} otherwise
78       */
79      boolean unsetField(String fieldName);
80  
81      /**
82       * Retrieve the value of a field from the event.
83       *
84       * @param fieldName The name of the field to retrieve.
85       * @return The value of the specified field
86       * @throws java.util.NoSuchElementException When the field is not part of the Event
87       */
88      Field getField(String fieldName);
89  
90      /**
91       * Retrieve a list of all fields (i.e. a list of field names) currently registered with the event.
92       * <p>
93       * Note that the returned implementation of {@link List} may or may not be thread safe. If multiple threads access
94       * the returned list concurrently, and at least one of the threads modifies the list structurally, it should be
95       * synchronized externally.
96       * <p>
97       * Note that the returned implementation of {@link List} may be an unmodifiable List (e.g. wrapped into
98       * {@code Collections.unmodifiableList()}.
99       *
100      * @return A list of field names
101      */
102     List<String> getFieldNames();
103 
104     /**
105      * Clears the content of the field's value.
106      * <p>
107      * This method calls the {@link Field#clear()} method of all fields associated with this event, and then removes
108      * them from the event. Use this call to wipe potentially confidential information from memory.
109      */
110     void clear();
111 
112     /**
113      * Create a JSON serialized representation of the event using the provided String encoding. This JSON
114      * representation in particular contains all fields that are registered with the event.
115      * <p>
116      * The event is serialized into a default format, the version of which can be obtained with the
117      * {@link Event#getRecordFormatVersion()} method.
118      * <p>
119      * When calling this method, make sure to use the same encoding that was used when setting the field values, for
120      * instance with the {@link Field#setValue(byte[], Encodings)} method.
121      *
122      * @param encoding The string encoding to use (e.g. UTF-8)
123      * @return A JSON representation of the event.
124      */
125     char[] toJson(String encoding);
126 }
