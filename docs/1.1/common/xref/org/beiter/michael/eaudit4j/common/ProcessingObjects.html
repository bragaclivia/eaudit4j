33  package org.beiter.michael.eaudit4j.common;
34  
35  
36  import org.apache.commons.lang3.Validate;
37  import org.slf4j.Logger;
38  import org.slf4j.LoggerFactory;
39  
40  import java.util.ArrayList;
41  import java.util.Collections;
42  import java.util.List;
43  import java.util.Map;
44  import java.util.NoSuchElementException;
45  import java.util.concurrent.ConcurrentHashMap;
46  
47  /**
48   * This class stores a set of processing objects, that is, objects that can be passed to a processor for use during
49   * event processing.
50   * <p>
51   * Examples for such object include, for instance, a data source object that can be used by a processor to persist
52   * data to a database.
53   */
54  // suppress warnings about the constructor (required for producing java docs)
55  @SuppressWarnings("PMD.UnnecessaryConstructor")
56  public class ProcessingObjects {
57  
58      /**
59       * The logger object for this class
60       */
61      private static final Logger LOG = LoggerFactory.getLogger(ProcessingObjects.class);
62  
63      /**
64       * A map that stores objects, identifying objects with a name.
65       */
66      private final Map<String, Object> objects = new ConcurrentHashMap<>();
67  
68      /**
69       * Constructs an empty set of processing objects.
70       * <p>
71       * Use the setter and getter to store objects and retrieve them.
72       */
73      public ProcessingObjects() {
74  
75          // no code here, constructor just for java docs
76      }
77  
78      /**
79       * Add an object to this structure.
80       * <p>
81       * Note that this method does <strong>not</strong> create a defensive copy of the provided {@code Object}).
82       *
83       * @param name  The name of the object to add
84       * @param value The object to add
85       * @return The name of the object that was added to the event
86       * @throws NullPointerException     When {@code name} or {@code value} is {@code null}
87       * @throws IllegalArgumentException When {@code name} is empty
88       */
89      public final String add(final String name, final Object value) {
90  
91          Validate.notBlank(name, "The validated character sequence 'name' is null or empty");
92          Validate.notNull(value, "The validated object 'object' is null");
93  
94          objects.put(name, value);
95  
96          return name;
97      }
98  
99      /**
100      * Checks if this structure contains an object with the specified name.
101      *
102      * @param name The object name to check for
103      * @return {@code true} if the this structure contains the object, {@code false} otherwise
104      */
105     public final boolean contains(final String name) {
106 
107         Validate.notBlank(name, "The validated character sequence 'name' is null or empty");
108 
109         return objects.containsKey(name);
110     }
111 
112     /**
113      * Remove an object from this struecture.
114      *
115      * @param name The name of the object to remove
116      * @return {@code true} if the object could be removed, {@code false} otherwise
117      * @throws NullPointerException     When {@code name} is {@code null}
118      * @throws IllegalArgumentException When {@code name} is empty
119      */
120     public final boolean remove(final String name) {
121 
122         Validate.notBlank(name, "The validated character sequence 'name' is null or empty");
123 
124         // return "true" if the field was found in the map, "false" otherwise
125         return objects.remove(name) != null;
126     }
127 
128     /**
129      * Retrieve an object from this structure.
130      * <p>
131      * Note that this method returns original objects, and does not make defensive copies.
132      *
133      * @param name The name of the object to retrieve.
134      * @return The object
135      * @throws NullPointerException     When {@code name} is {@code null}
136      * @throws IllegalArgumentException When {@code name} is empty
137      * @throws NoSuchElementException   When the object is not part of this structure
138      */
139     public final Object get(final String name) {
140 
141         Validate.notBlank(name, "The validated character sequence 'name' is null or empty");
142 
143         if (!objects.containsKey(name)) {
144             final String error = "The object " + name + " is not part of this structure.";
145             LOG.info(error);
146             throw new NoSuchElementException(error);
147         }
148 
149         // return the original (not a defensive copy!) of the requested object
150         return objects.get(name);
151     }
152 
153     /**
154      * Retrieve a list of all objects (i.e. a list of objects names) currently included in this structure.
155      * <p>
156      * Note that the returned implementation of {@link List} may or may not be thread safe. If multiple threads access
157      * the returned list concurrently, and at least one of the threads modifies the list structurally, it should be
158      * synchronized externally.
159      * <p>
160      * Note that the returned implementation of {@link List} may be an unmodifiable List (e.g. wrapped into
161      * {@code Collections.unmodifiableList()}.
162      *
163      * @return A list of field names
164      */
165     public final List<String> getObjectNames() {
166 
167         // Use an array list instead of a vector - much fast due to lack of synchronization,
168         // and the caller must cope with the implications of the List interface anyway...
169         final List<String> objectNames = new ArrayList<>(objects.keySet().size());
170 
171         // copy the object names into a new structure, we do not want to
172         // return the original enumeration that backs the internal map!
173         for (final String fieldName : objects.keySet()) {
174             objectNames.add(fieldName);
175         }
176 
177         // wrap the result to help the caller avoid accidental modifications...
178         return Collections.unmodifiableList(objectNames);
179     }
180 }
