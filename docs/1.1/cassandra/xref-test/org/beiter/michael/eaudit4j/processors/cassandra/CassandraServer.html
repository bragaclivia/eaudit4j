35  package org.beiter.michael.eaudit4j.processors.cassandra;
36  
37  import com.datastax.driver.core.Cluster;
38  import com.datastax.driver.core.Session;
39  import org.apache.thrift.transport.TTransportException;
40  import org.cassandraunit.utils.EmbeddedCassandraServerHelper;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  
44  import java.io.IOException;
45  import java.util.concurrent.atomic.AtomicBoolean;
46  
47  /**
48   * A utility class that starts an Cassandra server and populates it with test tables for auditing.
49   */
50  public class CassandraServer {
51  
52      private static final Logger LOG = LoggerFactory.getLogger(CassandraServer.class);
53  
54      public static final String HOST = "localhost";
55      public static final int PORT = 9142;
56      public static final String KEY_SPACE = "audit";
57  
58      private static AtomicBoolean serverRunning = new AtomicBoolean(false);
59      private static AtomicBoolean keyspaceCreated = new AtomicBoolean(false);
60      private static volatile Session session;
61      private static volatile Session sessionWithKeyspace;
62  
63      // prevent instantiation of this class
64      private CassandraServer() {
65      }
66  
67      /**
68       * Initializes the default schema and sets default values (if feasible). This deletes any data that are already in
69       * the database!
70       */
71      public static synchronized void init() {
72  
73          if (serverRunning.get()) {
74  
75              LOG.info("Initializing the Cassandra server with a default schema and default values");
76  
77              String[] createKeySpace = new String[]{
78                      // "USE " + KEY_SPACE + ";\n", -> rather specify the key space explicitly for the generic session object we use here
79                      "DROP TABLE IF EXISTS " + KEY_SPACE + ".events;\n",
80                      "CREATE TABLE " + KEY_SPACE + ".events (\n" +
81                              "  eventId     ASCII PRIMARY KEY,\n" +
82                              "  auditStream ASCII,\n" +
83                              "  eventJson   VARCHAR\n" +
84                              ");"
85              };
86  
87              try {
88                  for (String cql : createKeySpace) {
89                      session.execute(cql);
90                  }
91              } catch (Exception e) {
92                  LOG.error("The Cassandra server cannot be initialized", e.getCause(), e.getStackTrace());
93              }
94          } else {
95              LOG.info("The Cassandra server cannot be initialized because it is not running");
96          }
97  
98      }
99  
100     /**
101      * Starts the server, but does not perform any database schema initialization
102      *
103      * @throws Exception When the server cannot be started
104      */
105     public static void start()
106             throws Exception {
107 
108         LOG.info("Checking if the Cassandra server can be started");
109 
110         // start the server only if there is not already a server running
111         if (serverRunning.compareAndSet(false, true)) {
112 
113             try {
114                 LOG.info("Starting the Cassandra server");
115                 EmbeddedCassandraServerHelper.startEmbeddedCassandra(20000L);
116                 Cluster cluster = new Cluster.Builder()
117                         .addContactPoint(HOST)
118                         .withPort(PORT)
119                         .build();
120 
121                 session = cluster.connect();
122 
123                 // create the key space
124                 LOG.info("Creating the Cassandra key space");
125                 String[] createDb = new String[]{
126                         "DROP KEYSPACE IF EXISTS " + KEY_SPACE + ";\n",
127                         "CREATE KEYSPACE " + KEY_SPACE + " WITH REPLICATION = {'class': 'SimpleStrategy', 'replication_factor' : 1};\n",
128                 };
129 
130                 try {
131                     for (String cql : createDb) {
132                         session.execute(cql);
133                     }
134                 } catch (Exception e) {
135                     LOG.error("The keyspace cannot be created on the Cassandra server", e.getCause(), e.getStackTrace());
136                 }
137 
138                 sessionWithKeyspace = cluster.connect(KEY_SPACE);
139                 keyspaceCreated.set(true);
140 
141                 LOG.info("Cassandra server started");
142             } catch (IOException | TTransportException | InterruptedException e) {
143                 serverRunning.set(false);
144                 LOG.warn("Cassandra server not started due to error");
145                 throw new Exception(e);
146             }
147         } else {
148             LOG.info("Cassandra server already running");
149         }
150     }
151 
152     /**
153      * Stops the server
154      */
155     public static void stop() {
156 
157         LOG.info("Checking if the Cassandra server can be stopped");
158 
159         // stop the server only if there is a server running
160         if (serverRunning.compareAndSet(true, false)) {
161 
162             LOG.info("Stopping the Cassandra server");
163 
164             // clear all data from the cluster
165             EmbeddedCassandraServerHelper.cleanEmbeddedCassandra();
166 
167             // close the cluster and all sessions, and shutdown the embedded server
168             session.getCluster().close();
169 
170             LOG.info("Cassandra server stopped");
171         } else
172             LOG.info("Cassandra server not running");
173     }
174 
175 
176     /**
177      * Restarts the server, but does not perform any database schema initialization
178      *
179      * @throws Exception When the server cannot be started
180      */
181     public static void restart()
182             throws Exception {
183 
184         stop();
185         start();
186     }
187 
188     /**
189      * Returns a session on the Cassandra cluster if the cluster is up and running (not closed).
190      * <p>
191      * The session is not bound to a specific keyspace. Use the "getSessionWithKeyspace()" method to get a keyspaced
192      * session that is specific to the keyspace used in these unit tests.
193      * <p>
194      * See http://www.datastax.com/dev/blog/4-simple-rules-when-using-the-datastax-drivers-for-cassandra
195      *
196      * @return A session on the cluster
197      * @throws IllegalStateException when the cluster is not running
198      */
199     public static Session getSession() {
200 
201         if (serverRunning.get() && !session.isClosed()) {
202             return session;
203         } else {
204             if (session.isClosed()) {
205                 throw new IllegalStateException("Session is closed");
206             } else {
207                 throw new IllegalStateException("Cluster not running");
208             }
209         }
210     }
211 
212     /**
213      * Returns a keyspaced session on the Cassandra cluster if the cluster is up and running (not closed) AND the
214      * keyspace has been created / initialized.
215      * <p>
216      * The session is bound to the keyspace used in these unit tests.
217      * <p>
218      * See http://www.datastax.com/dev/blog/4-simple-rules-when-using-the-datastax-drivers-for-cassandra
219      *
220      * @return A session on the cluster
221      * @throws IllegalStateException when the cluster is not running
222      */
223     public static Session getSessionWithKeyspace() {
224 
225         if (serverRunning.get() && keyspaceCreated.get() && !sessionWithKeyspace.isClosed()) {
226             return sessionWithKeyspace;
227         } else {
228             if (sessionWithKeyspace.isClosed()) {
229                 throw new IllegalStateException("Session is closed");
230             } else {
231                 throw new IllegalStateException("Cluster not running or keyspace has not been created");
232             }
233         }
234     }
