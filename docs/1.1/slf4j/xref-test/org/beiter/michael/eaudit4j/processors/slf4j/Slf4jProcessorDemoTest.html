56  public class Slf4jProcessorDemoTest {
57  
58      @Test
59      public void slf4JProcessorDemoTest()
60              throws FactoryException, AuditException, UnsupportedEncodingException {
61  
62          // see the provided log4j.properties for an example on how to configure log4j to log these MDC fields
63          Map<String, String> props = new HashMap<>();
64          props.put(MapBasedSlf4jPropsBuilder.KEY_MDC_FIELDS,"subject,actor:myActor,byteField,invalidField:neverExists");
65  
66          CommonProperties properties = MapBasedCommonPropsBuilder.build(props);
67          properties.setDefaultAuditStream("1234567890");
68          properties.setEncoding("UTF-8");
69          properties.setFailOnMissingProcessors(true);
70          properties.setProcessors(Slf4jProcessor.class.getCanonicalName());
71  
72          Audit audit = AuditFactory.getInstance(properties.getAuditClassName(), properties);
73  
74          Field field = new EventField("byteField", new Hex().encode("1234".getBytes("UTF-8")), Encodings.HEX);
75  
76          // create the event, using (amongst others) the custom field above
77          Event event = new EventBuilder(properties)
78                  .setSubject("SubjectId-1234".toCharArray())
79                  .setObject("ObjectId-3456".toCharArray())
80                  .setActor("ActorId-5678".toCharArray())
81                  .setResult("Some result".toCharArray())
82                  .setField(field)
83                  .build();
84  
85          audit.audit(event);
86  
87      }
