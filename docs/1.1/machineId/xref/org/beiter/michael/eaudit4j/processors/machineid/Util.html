36  package org.beiter.michael.eaudit4j.processors.machineid;
37  
38  import org.beiter.michael.eaudit4j.common.AuditErrorConditions;
39  import org.beiter.michael.eaudit4j.common.AuditException;
40  import org.beiter.michael.eaudit4j.processors.machineid.propsbuilder.MapBasedMachineIdPropsBuilder;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  
44  import java.net.InetAddress;
45  import java.net.UnknownHostException;
46  import java.util.Date;
47  import java.util.UUID;
48  
49  /**
50   * A utility class providing methods to resolve the machine ID from:
51   * <ul>
52   * <li>an environment variable or</li>
53   * <li>a hostname or</li>
54   * <li>as a random ID</li>
55   * </ul>
56   */
57  @SuppressWarnings("PMD.ShortClassName")
58  public final class Util {
59  
60      /**
61       * The logger object for this class
62       */
63      private static final Logger LOG = LoggerFactory.getLogger(Util.class);
64  
65      /**
66       * A private constructor to prevent instantiation of this class
67       */
68      private Util() {
69      }
70  
71      /**
72       * This method retrieves the machine ID as configured in a specific environment variable.
73       * <p>
74       * If the env variable is not configured or empty, the host ID defaults to "$HOSTNAME:$TIMESTAMP", where:
75       * <ul>
76       * <li>$HOSTNAME is the fully qualified hostname of localhost (see {@link InetAddress#getLocalHost()} and
77       * {@link InetAddress#getCanonicalHostName()})</li>
78       * <li>$TIMESTAMP is the epoch timestamp</li>
79       * </ul>
80       * <p>
81       * If the hostname cannot be resolved from the environment or the hostname, then a ID is generated as the
82       * machine ID.
83       *
84       * @param getfromEnv      True if the method should try reading the machine ID from the environment, false to skip
85       * @param envVariableName The name of the environment variable to read the machine ID from
86       * @param getFromHostname True if the method should try crafting the machine ID from the hostname, false to skip
87       * @return The machine ID as resolved by the algorithm described above
88       * @throws AuditException When the machineID cannot be determined
89       */
90      // Allow catching a general Exception because of possible RuntimeException thrown by getCanonicalHostName(), which
91      // we will ignore and are handling in business logic as part of the flow (i.e. we will do an alternative resolve)
92      @SuppressWarnings("PMD.AvoidCatchingGenericException")
93      public static String getMachineId(final boolean getfromEnv,
94                                        final String envVariableName,
95                                        final boolean getFromHostname)
96              throws AuditException {
97  
98          // this is the machine ID that we will eventually use as a return value of this method
99          String resolvedMachineId = null;
100 
101         // check whether we should read this Machine ID from an environment variable, and do so if necessary
102         if (getfromEnv) {
103 
104             LOG.debug("Trying to obtain the machine ID from the environment...");
105 
106             resolvedMachineId = getMachineIdFromEnv(envVariableName);
107         }
108 
109         // check whether we should try resolve the hostname and craft a machine ID from that
110         if (getFromHostname) {
111 
112             LOG.debug("Trying to obtain the machine ID from the hostname...");
113 
114             try {
115                 resolvedMachineId = getCanonicalHostName() + ":" + new Date().getTime();
116             } catch (Exception e) {
117                 // Catching general Exception because of possible RuntimeException thrown by getCanonicalHostName()
118                 final String error = "Canonical hostname can not be resolved: " + e.getLocalizedMessage()
119                         + " Attempting alternative machine ID resolution method.";
120 
121                 LOG.info(error, e);
122             }
123         }
124 
125         // Fallback: if we failed to resolve the machine ID up to this point, and we also did not bail out yet with an
126         //           exception, then we create a random string and use that as the machine ID, logging this in the app
127         //           log so that the operator can find it and properly map the audit logs to a specific machine
128 
129         if (resolvedMachineId == null || resolvedMachineId.isEmpty()) {
130 
131             LOG.warn("Generating a random machine ID");
132 
133             resolvedMachineId = UUID.randomUUID().toString();
134 
135             // create a log trace so that the operator can see what machine ID is going to be used:
136             LOG.warn(MapBasedMachineIdPropsBuilder.KEY_MACHINE_ID + " has been set to: " + resolvedMachineId);
137         }
138 
139         return resolvedMachineId;
140     }
141 
142     /**
143      * Retrieve the machine ID a provided environment variable
144      *
145      * @param envVariableName The environment variable that holds the machine ID
146      * @return The machine ID (as stored in the env variable)
147      * @throws AuditException When the provided environment variable name is null or empty
148      */
149     private static String getMachineIdFromEnv(final String envVariableName)
150             throws AuditException {
151 
152         if (envVariableName == null || envVariableName.isEmpty()) {
153             final String error = "The configured Machine ID environment variable name is null or empty";
154             LOG.warn(error);
155             throw new AuditException(AuditErrorConditions.CONFIGURATION, error);
156         }
157 
158         final String resolvedMachineId = System.getenv(envVariableName);
159 
160         if (resolvedMachineId == null || resolvedMachineId.isEmpty()) {
161             final String error = "The value of the configured Machine ID environment variable is null or empty. "
162                     + "Attempting alternative machine ID resolution method.";
163             LOG.warn(error);
164         }
165         return resolvedMachineId;
166     }
167 
168 
169     /**
170      * This method tries to retrieve the canonical machine name.
171      *
172      * @return the canonical machine name
173      * @throws AuditException When the hostname cannot be determined
174      */
175     private static String getCanonicalHostName()
176             throws AuditException {
177 
178         String canonicalHostName;
179         try {
180             canonicalHostName = InetAddress.getLocalHost().getCanonicalHostName();
181         } catch (UnknownHostException e) {
182             final String error = "Error when retrieving the canonical host name: " + e.getLocalizedMessage();
183             LOG.warn(error, e);
184             throw new AuditException(AuditErrorConditions.INITIALIZATION, error, e);
185         }
186 
187         return canonicalHostName;
188     }
