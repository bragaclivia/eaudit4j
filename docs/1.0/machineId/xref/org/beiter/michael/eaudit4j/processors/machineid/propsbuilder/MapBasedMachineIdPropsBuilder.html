141     /**
142      * Initialize a set of machine ID properties based on key / values in a <code>HashMap</code>.
143      *
144      * @param properties A <code>HashMap</code> with configuration properties, using the keys as specified in this class
145      * @return A {@link MachineIdProperties} object with default values, plus the provided parameters
146      * @throws NullPointerException When {@code properties} is {@code null}
147      */
148     // CHECKSTYLE:OFF
149     // this is flagged in checkstyle with a missing whitespace before '}', which is a bug in checkstyle
150     // suppress warnings about this method being too long (not much point in splitting up this one!)
151     // suppress warnings about this method being too complex (can't extract a generic subroutine to reduce exec paths)
152     @SuppressWarnings({"PMD.ExcessiveMethodLength", "PMD.NPathComplexity", "PMD.CyclomaticComplexity", "PMD.StdCyclomaticComplexity", "PMD.ModifiedCyclomaticComplexity"})
153     // CHECKSTYLE:ON
154     public static MachineIdProperties build(final Map<String, String> properties) {
155 
156         Validate.notNull(properties, "The validated object 'value' is null");
157 
158         final MachineIdProperties machineIdProperties = new MachineIdProperties();
159         String tmp = properties.get(KEY_MACHINE_ID);
160         if (StringUtils.isNotEmpty(tmp)) {
161             machineIdProperties.setMachineId(tmp);
162             logValue(KEY_MACHINE_ID, tmp);
163         } else {
164             machineIdProperties.setMachineId(DEFAULT_MACHINE_ID);
165             logDefault(KEY_MACHINE_ID, DEFAULT_MACHINE_ID);
166         }
167 
168         tmp = properties.get(KEY_EVENT_FIELD_NAME);
169         if (StringUtils.isNotEmpty(tmp)) {
170             machineIdProperties.setEventFieldName(tmp);
171             logValue(KEY_EVENT_FIELD_NAME, tmp);
172         } else {
173             machineIdProperties.setEventFieldName(DEFAULT_EVENT_FIELD_NAME);
174             logDefault(KEY_EVENT_FIELD_NAME, DEFAULT_EVENT_FIELD_NAME);
175         }
176 
177         tmp = properties.get(KEY_MACHINE_ID_FROM_ENV);
178         if (StringUtils.isNotEmpty(tmp)) {
179             machineIdProperties.setMachineIdFromEnv(Boolean.parseBoolean(tmp));
180             logValue(KEY_MACHINE_ID_FROM_ENV, tmp);
181         } else {
182             machineIdProperties.setMachineIdFromEnv(DEFAULT_MACHINE_ID_FROM_ENV);
183             logDefault(KEY_MACHINE_ID_FROM_ENV, String.valueOf(DEFAULT_MACHINE_ID_FROM_ENV));
184         }
185 
186         tmp = properties.get(KEY_MACHINE_ID_ENV_NAME);
187         if (StringUtils.isNotEmpty(tmp)) {
188             machineIdProperties.setMachineIdEnvName(tmp);
189             logValue(KEY_MACHINE_ID_ENV_NAME, tmp);
190         } else {
191             machineIdProperties.setMachineIdEnvName(DEFAULT_MACHINE_ID_ENV_NAME);
192             logDefault(KEY_MACHINE_ID_ENV_NAME, DEFAULT_MACHINE_ID_ENV_NAME);
193         }
194 
195         tmp = properties.get(KEY_MACHINE_ID_FROM_HOSTNAME);
196         if (StringUtils.isNotEmpty(tmp)) {
197             machineIdProperties.setMachineIdFromHostname(Boolean.parseBoolean(tmp));
198             logValue(KEY_MACHINE_ID_FROM_HOSTNAME, tmp);
199         } else {
200             machineIdProperties.setMachineIdFromHostname(DEFAULT_MACHINE_ID_FROM_HOSTNAME);
201             logDefault(KEY_MACHINE_ID_FROM_HOSTNAME, String.valueOf(DEFAULT_MACHINE_ID_FROM_HOSTNAME));
202         }
203 
204         // set the additional properties, preserving the originally provided properties
205         // create a defensive copy of the map and all its properties
206         // the code looks a little complicated that "putAll()", but it catches situations where a Map is provided that
207         // supports null values (e.g. a HashMap) vs Map implementations that do not (e.g. ConcurrentHashMap).
208         final Map<String, String> tempMap = new ConcurrentHashMap<>();
209         for (final Map.Entry<String, String> entry : properties.entrySet()) {
210             final String key = entry.getKey();
211             final String value = entry.getValue();
212 
213             if (value != null) {
214                 tempMap.put(key, value);
215             }
216         }
217         machineIdProperties.setAdditionalProperties(tempMap);
218 
219         return machineIdProperties;
220     }

222     /**
223      * Create a log entry when a value has been successfully configured.
224      *
225      * @param key   The configuration key
226      * @param value The value that is being used
227      */
228     private static void logValue(final String key, final String value) {
229 
230         // Fortify will report a violation here because of disclosure of potentially confidential information.
231         // However, the configuration keys are not confidential, which makes this a non-issue / false positive.
232         if (LOG.isInfoEnabled()) {
233             final StringBuilder msg = new StringBuilder("Key found in configuration ('")
234                     .append(key)
235                     .append("'), using configured value (not disclosed here for security reasons)");
236             LOG.info(msg.toString());
237         }
238 
239         // Fortify will report a violation here because of disclosure of potentially confidential information.
240         // The configuration VALUES are confidential. DO NOT activate DEBUG logging in production.
241         if (LOG.isDebugEnabled()) {
242             final StringBuilder msg = new StringBuilder("Key found in configuration ('")
243                     .append(key)
244                     .append("'), using configured value ('");
245             if (value == null) {
246                 msg.append("null')");
247             } else {
248                 msg.append(value).append("')");
249             }
250             LOG.debug(msg.toString());
251         }
252     }

254     /**
255      * Create a log entry when a default value is being used in case the propsbuilder key has not been provided in the
256      * configuration.
257      *
258      * @param key          The configuration key
259      * @param defaultValue The default value that is being used
260      */
261     private static void logDefault(final String key, final String defaultValue) {
262 
263         // Fortify will report a violation here because of disclosure of potentially confidential information.
264         // However, neither the configuration keys nor the default propsbuilder values are confidential, which makes
265         // this a non-issue / false positive.
266         if (LOG.isInfoEnabled()) {
267             final StringBuilder msg = new StringBuilder("Key is not configured ('")
268                     .append(key)
269                     .append("'), using default value ('");
270             if (defaultValue == null) {
271                 msg.append("null')");
272             } else {
273                 msg.append(defaultValue).append("')");
274             }
275             LOG.info(msg.toString());
276         }
277     }
