56  /**
57   * This processors allows to either configure a machine identifier through
58   * a configuration property, or to determine it from an environment variable,
59   * from the hostname, or fall back on a randomly generated ID.
60   * <p>
61   * Note that a specific instance of this class will always return the same
62   * machine ID once it has been determined.
63   */
64  public class MachineIdProcessor
65          implements Processor, Reversible {
66  
67      /**
68       * The logger object for this class
69       */
70      private static final Logger LOG = LoggerFactory.getLogger(MachineIdProcessor.class);
71  
72      /**
73       * A copy of the common properties
74       */
75      private CommonProperties commonProperties;
76  
77      /**
78       * A copy of the processor specific properties
79       */
80      private MachineIdProperties properties;
81  
82      /**
83       * The machine ID
84       * (not static, because we want applications to use more than one machine ID configuration if they need to)
85       */
86      private final AtomicReference<String> machineId = new AtomicReference<>();
87  
88  
89      /**
90       * {@inheritDoc}
91       */
92      @Override
93      public final void init(final CommonProperties pCommonProperties) {
94  
95          Validate.notNull(pCommonProperties, "The validated object 'pCommonProperties' is null");
96  
97          this.commonProperties = new CommonProperties(pCommonProperties);
98          this.properties = MapBasedMachineIdPropsBuilder.build(pCommonProperties.getAdditionalProperties());
99      }

101     /**
102      * This method processes an event in the default audit stream.
103      * <p>
104      * The method obtains a machine ID, and adds that machine ID to the event.
105      * <p>
106      * See {@link org.beiter.michael.eaudit4j.common.Audit#audit(Event)}.
107      * See {@link MachineIdProcessor#process(Event, String, ProcessingObjects)}.
108      *
109      * @param event The event to audit
110      * @return The event after processing (for synchronous audit implementations), undefined otherwise
111      * @throws AuditException when the audit operation fails (e.g. no machine ID could be obtained)
112      */
113     @Override
114     public final Event process(final Event event)
115             throws AuditException {
116 
117         return process(event, commonProperties.getDefaultAuditStream());
118     }

120     /**
121      * This method processes an event in the provided audit stream.
122      * <p>
123      * The method obtains a machine ID, and adds that machine ID to the event.
124      * <p>
125      * See {@link org.beiter.michael.eaudit4j.common.Audit#audit(Event)}.
126      * See {@link MachineIdProcessor#process(Event, String, ProcessingObjects)}.
127      *
128      * @param event           The event to audit
129      * @param auditStreamName The audit stream to send events to
130      * @return The event after processing (for synchronous audit implementations), undefined otherwise
131      * @throws AuditException           when the audit operation fails (e.g. no machine ID could be obtained)
132      * @throws NullPointerException     When the {@code auditStreamName} or {@code processingObjects} are {@code null}
133      * @throws IllegalArgumentException When {@code auditStreamName} is empty
134      */
135     @Override
136     public final Event process(final Event event, final String auditStreamName)
137             throws AuditException {
138 
139         return process(event, auditStreamName, new ProcessingObjects());
140     }

142     /**
143      * This method processes an event in the provided audit stream and includes a set of {@link ProcessingObjects}.
144      * <p>
145      * The method obtains a machine ID, and adds that machine ID to the event.
146      * <p>
147      * The machine ID is resolved as follows:
148      * <ul>
149      * <li>First, check if a machine ID has been explicitly configured, and use that ID</li>
150      * <li>If no machine ID has been configured, try to retrieve the machine ID from an environment variable.</li>
151      * <li>If this failed (e.g. the processor is not configured to retrieve the machine ID from the environment,
152      * or that retrieval operation failed), then try build a machine ID from a combination of the canonical host name
153      * and a timestamp</li>
154      * <li>If this failed (e.g. the processor is not configured to use the canonical hostname, or the canonical
155      * hostname could not be resolved), then the processor will create a pseudo-random machine ID and log that
156      * machine ID in the log.</li>
157      * </ul>
158      * <p>
159      * See {@link org.beiter.michael.eaudit4j.common.Audit#audit(Event, String)}.
160      *
161      * @param event             The event to audit
162      * @param auditStreamName   The audit stream to send events to
163      * @param processingObjects The processing objects available to the processors
164      * @return The event after processing (for synchronous audit implementations), undefined otherwise
165      * @throws AuditException           when the audit operation fails (e.g. the event already contains a machine ID,
166      *                                  or no machine ID could be obtained)
167      * @throws NullPointerException     When the {@code auditStreamName} or {@code processingObjects} are {@code null}
168      * @throws IllegalArgumentException When {@code auditStreamName} is empty
169      */
170     @Override
171     public final Event process(final Event event,
172                                final String auditStreamName,
173                                final ProcessingObjects processingObjects)
174             throws AuditException {
175 
176         if (event == null) {
177             final String error = "The validated object 'event' is null";
178             LOG.warn(error);
179             throw new AuditException(AuditErrorConditions.INVALID_EVENT, error);
180         }
181 
182         // this implementation does not use audit streams, but validate anyway to ensure library API compliance
183         Validate.notBlank(auditStreamName, "The validated character sequence 'auditStreamName' is null or empty");
184 
185         // this implementation does not use processing objects, but validate anyway to ensure library API compliance
186         Validate.notNull(processingObjects, "The validated object 'processingObjects' is null");
187 
188         // Make sure that the properties we need are available
189         if (properties == null) {
190 
191             final String error = "eAudit4j processor specific properties for " + this.getClass().getCanonicalName()
192                     + " have not been initialized.";
193             LOG.warn(error);
194             throw new AuditException(AuditErrorConditions.INITIALIZATION, error);
195         }
196 
197         // throw an exception if the event already contains a machine ID.
198         // At best, this is an indication that this processor has been executed repeatedly in the audit chain
199         if (event.containsField(properties.getEventFieldName())) {
200 
201             final String error = "The processor " + this.getClass().getCanonicalName()
202                     + " is present multiple times in the audit chain";
203             LOG.warn(error);
204             throw new AuditException(AuditErrorConditions.CONFIGURATION, error);
205         }
206 
207         // we could further reduce unnecessary performance loss by introducing a double synchronized block, but it is
208         // okay to go through the machine ID resolution multiple times (and throw n-1 of the results away) like this
209         // because the operation is cheap and we can so avoid unnecessary locking through a double-check if/synchronized
210         if (machineId.get() == null) {
211 
212             String resolvedMachineId;
213 
214             // if the properties have a machine ID configured, we can use that instead of making one up / creating one
215             // from other means:
216             if (properties.getMachineId() == null || properties.getMachineId().isEmpty()) {
217 
218                 resolvedMachineId = Util.getMachineId(
219                         properties.isMachineIdFromEnv(),
220                         properties.getMachineIdEnvName(),
221                         properties.isMachineIdFromHostname());
222             } else {
223 
224                 resolvedMachineId = properties.getMachineId();
225             }
226 
227             machineId.compareAndSet(null, resolvedMachineId);
228         }
229 
230         // add the machine ID to the event
231         byte[] bytes;
232         try {
233             bytes = machineId.get().getBytes(commonProperties.getEncoding());
234         } catch (UnsupportedEncodingException e) {
235             final String error = "Encoding is not supported: " + commonProperties.getEncoding();
236             LOG.warn(error);
237             throw new AuditException(AuditErrorConditions.PROCESSING, error, e);
238         }
239 
240         final Field field = new EventField(properties.getEventFieldName(), bytes);
241 
242         event.setField(field);
243 
244         // clear confidential data from local variables
245         Cleanser.wipe(bytes);
246         field.clear();
247 
248         return event;
249     }

251     /**
252      * This processor does not store any confidential information.
253      * <p>
254      * The implementation of this method does nothing.
255      */
256     @Override
257     public void cleanUp() {
258 
259         // do nothing
260     }
261 
262     /**
263      * This method removes the machine ID field from an event. If the event does not contain the field, then it is
264      * returned unchanged.
265      *
266      * @param event The event to revert changes on
267      * @return An event with the machine ID field removed
268      * @throws AuditException When the operation fails
269      */
270     @Override
271     public final Event revert(final Event event)
272             throws AuditException {
273 
274         Validate.notNull(event, "The validated object 'event' is null");
275 
276         if (event.containsField(properties.getEventFieldName())) {
277 
278             final boolean result = event.unsetField(properties.getEventFieldName());
279 
280             if (result) {
281                 return event;
282             } else {
283                 final String error = "The machine ID field could not be removed from the event";
284                 LOG.warn(error);
285                 throw new AuditException(AuditErrorConditions.PROCESSING, error);
286             }
287         } else {
288 
289             return event;
290         }
291     }
