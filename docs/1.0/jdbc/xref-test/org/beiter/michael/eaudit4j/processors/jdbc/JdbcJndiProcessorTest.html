158     /**
159      * Test that we can retrieve a connection.
160      */
161     @Test
162     public void getConnectionTest()
163             throws InvocationTargetException, IllegalAccessException, SQLException {
164 
165         Map<String, String> props = new HashMap<>();
166 
167         // the JDBC Processor configuration
168         props.put(MapBasedJdbcPropsBuilder.KEY_JNDI_CONNECTION_NAME, JNDI_NAME);
169 
170         JdbcProperties jdbcProps = MapBasedJdbcPropsBuilder.build(props);
171         CommonProperties commonProps = MapBasedCommonPropsBuilder.build(props);
172         Processor processor = new JdbcJndiProcessor();
173         processor.init(commonProps);
174 
175         Connection con = (Connection) method_getConnection.invoke(processor, jdbcProps, new ProcessingObjects());
176 
177         String error = "The connection is null";
178         assertThat(error, con, is(notNullValue()));
179 
180         con.close();
181     }

187     /**
188      * Test that an invalid JNDI name throws an exception
189      */
190     @Test(expected = AuditException.class)
191     public void invalidJndiNameTest()
192             throws FactoryException, AuditException, UnsupportedEncodingException, SQLException {
193 
194         Map<String, String> props = new HashMap<>();
195 
196         // the JDBC Processor configuration
197         props.put(MapBasedJdbcPropsBuilder.KEY_JNDI_CONNECTION_NAME, JNDI_NAME + "_INVALID");
198         props.put(MapBasedJdbcPropsBuilder.KEY_EVENT_ID_FIELD_NAME, EVENT_ID_FIELD_NAME);
199         props.put(MapBasedJdbcPropsBuilder.KEY_INSERT_EVENT_SQL_STMT,
200                 "INSERT INTO events (eventId, auditStreamName, eventJson) VALUES (?, ?, ?)");
201         props.put(MapBasedJdbcPropsBuilder.KEY_INDEXED_FIELDS, "");
202 
203         // the audit library common configuration
204         CommonProperties properties = MapBasedCommonPropsBuilder.build(props);
205         properties.setDefaultAuditStream(T_AUDIT_STREAM_NAME);
206         properties.setEncoding("UTF-8");
207         properties.setFailOnMissingProcessors(true);
208         properties.setProcessors(JdbcJndiProcessor.class.getCanonicalName());
209 
210         Audit audit = AuditFactory.getInstance(properties.getAuditClassName(), properties);
211 
212         Event event = getTestEvent(properties);
213 
214         audit.audit(event);
215     }

238     /**
239      * Produce a test event
240      */
241     private Event getTestEvent(CommonProperties properties) throws UnsupportedEncodingException {
242 
243         // the event ID is required for this processor to work
244         Field eventIdField = new EventField(EVENT_ID_FIELD_NAME, EVENT_ID.getBytes("UTF-8"));
245 
246         // test data
247         Field field = new EventField("byteField", new Hex().encode("1234".getBytes("UTF-8")), Encodings.HEX);
248 
249         // create the event, using (amongst others) the custom field above
250         return new EventBuilder(properties)
251                 .setField(eventIdField)
252                 .setSubject(T_SUBJECT.toCharArray())
253                 .setObject(T_OBJECT.toCharArray())
254                 .setActor(T_ACTOR.toCharArray())
255                 .setResult(T_RESULT.toCharArray())
256                 .setField(field)
257                 .build();
258     }
