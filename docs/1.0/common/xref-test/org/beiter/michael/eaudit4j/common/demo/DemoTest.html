76      /**
77       * This is how to use the library when creating events with the "Event" interface constructor and Field setter.
78       * <p>
79       * This is the most basic way to use the library. The direct creation of fields and usage of the basic {@link Event}
80       * interface only allows you to create "unnamed" fields directly, but you can create "named" fields by using the
81       * {@link EventField} constructor.
82       * <p>
83       * This is the most flexible way to use the audit library, because you are not restricted in any way by the
84       * library's interface assumptions on field names to use in your application. You can use this approach to create
85       * application specific wrappers and customized "convenience" methods to make calls to the audit subsystem.
86       * <p>
87       * Note that this approach is using fairly low-level APIs of the audit library, which makes it necessary to handle
88       * the conversion of audit payload to binary data (which the library processes) yourself.
89       */
90      @Test
91      public void auditWithAuditLibraryPrimitivesTest()
92              throws UnsupportedEncodingException {
93  
94          // First create some fields we want to audit
95          // Note that the field content is submitted as a binary value (see other options to use human readable formats)
96          Field field1 = new EventField("field_1_name", "field 1 value".getBytes(encoding));
97          // Fields may have multi-byte chars, of course:
98          Field field2 = new EventField("field_2_name", "field 2 value: \u00C4-\u00D6-\u00DC".getBytes(encoding));
99          // You can audit binary data, too.
100         // The examples assume some binary data that has been encoded in its byte representation:
101         Field field3 = new EventField(
102                 "field_3_name",
103                 Base64.encodeBase64("field 3 value".getBytes(encoding)),
104                 Encodings.BASE64);
105         Field field4 = new EventField(
106                 "field_4_name",
107                 new Hex().encode("field 4 value".getBytes(encoding)),
108                 Encodings.HEX);
109 
110         // The library defines some "common" (standardized) field names in the properties (which you can override
111         // there), that  you can use like so:
112         Field field5 = new EventField(
113                 properties.getFieldNameActor(),
114                 "actor ID".getBytes(encoding));
115         Field field6 = new EventField(
116                 properties.getFieldNameSubject(),
117                 "subject ID".getBytes(encoding));
118 
119         // create an event, and set some fields in the constructor (optional)
120         Event event = new AuditEvent(field1, field2, field3);
121         // you may set more fields with the Field setter method
122         event.setField(field4);
123         event.setField(field5);
124         event.setField(field6);
125 
126         // this is where you pass the event into the audit processing chain
127         auditEvent(event);
128 
129         // Clear the confidential information from the event
130         // (if you always want to do that, feel free to move this to the {@code auditEvent()} method.)
131         event.clear();
132 
133         // Clear the confidential information from the fields
134         field1.clear();
135         field2.clear();
136         field3.clear();
137         field4.clear();
138         field5.clear();
139         field6.clear();
140     }

142     /**
143      * This is how to use the library when creating events with the "ExtendedEvent" interface constructor and advanced
144      * Field setters.
145      * <p>
146      * This is a more advanced way to use the library. Using the {@link ExtendedEvent} interface allows you to create
147      * commonly used / "named" fields directly.
148      * <p>
149      * Unless you want to audit custom fields for your application, you will not have to create {@link Field}s directly
150      * and you will hence not have to worry about proper encoding of binary data.
151      */
152     @Test
153     public void auditWithAuditLibraryExtensionsTest()
154             throws UnsupportedEncodingException {
155 
156         // no need to create fields before creating the event unless you want to add some custom
157         // fields (including binary data, see auditWithAuditLibraryPrimitivesTest()), like so:
158         Field field1 = new EventField("a_custom_field_1_name", "field 1 value".getBytes(encoding));
159         // Fields may have multi-byte chars, of course:
160         Field field2 = new EventField("a_custom_field_2_name",
161                 "field 2 value: \u00C4-\u00D6-\u00DC".getBytes(encoding));
162 
163         // If you are not using custom fields, you may use the default constructor and add data to the event using the
164         // named setters. You will have to pass in the application properties, because the ExtendedAuditEvent is more
165         // sophisticated than the basic AuditEvent and needs some configuration.
166         // You can additionally still use the Field constructor, and the generic Field setter, to add custom events,
167         // like so:
168         ExtendedEvent event = new ExtendedAuditEvent(properties, field1);
169         // the generic setter for the custom field
170         event.setField(field2);
171         // named setters for commonly used fields
172         event.setActor("actor ID".toCharArray());
173         event.setSubject("subject ID".toCharArray());
174 
175 
176         // If you do not need custom fields, creating and populating an event can be as easy as:
177         ExtendedEvent event2 = new ExtendedAuditEvent(properties);
178         event.setActor("actor ID".toCharArray());
179         event.setSubject("subject ID".toCharArray());
180 
181         // this is where you pass the event into the audit processing chain
182         auditEvent(event);
183         auditEvent(event2);
184 
185         // Clear the confidential information from the event
186         // (if you always want to do that, feel free to move this to the {@code auditEvent()} method.)
187         event.clear();
188         event2.clear();
189 
190         // Clear the confidential information from the fields
191         field1.clear();
192         field2.clear();
193     }

195     /**
196      * When you are mostly using the "common" fields in your events, but you do not want to bother with the difference
197      * between {@link Event}s and {@link ExtendedEvent}s, then you may want to use the {@link EventBuilder}.
198      * <p>
199      * The {@link EventBuilder} is a convenient way to create {@link Event}s and populate the "commonly used" fields,
200      * but still being able to add "custom fields":
201      * <ul>
202      *      <li>Before invocation of the event builder (create a {@link Field} and add it to the event builder)</li>
203      *      <li>During invocation of the event builder (specify custom field information, including binary payload, to the event builder)</li>
204      *      <li>After invocation of the event builder (create a {@link Field} and add it to the event that has been created by the event builder)</li>
205      * </ul>
206      * {@link Event#setField(Field)} method.
207      * <p>
208      */
209     @Test
210     public void auditWithAuditEventBuilderTest()
211             throws UnsupportedEncodingException {
212 
213         // The event builder can use some pre-created fields
214         Field field1 = new EventField("a_custom_field_1_name", "field 1 value".getBytes(encoding));
215 
216         // The event builder will need access to the configuration properties
217         Event event = new EventBuilder(properties)
218                 .setActor("actor ID".toCharArray())
219                 .setSubject("subject ID".toCharArray())
220                 .setField(field1) // use a pre-created field
221                 .setField("a_custom_field_3_name", "field 3 value".getBytes(encoding)) // use a custom field, with binary data
222                 .setField("a_custom_field_4_name", "field 4 value".toCharArray()) // use a custom field, with char data
223                 .setField(properties.getFieldNameObject(), "field 5 value".getBytes(encoding)) // use a custom field (named), with binary data
224                 .setField(properties.getFieldNameResult(), "field 6 value".toCharArray()) // use a custom field (named), with binary data
225                 .build();
226 
227         // you can optionally still use the primitives with the Event created by the builder to add more custom fields,
228         // including binary data (see auditWithAuditLibraryPrimitivesTest())...
229         Field field2 = new EventField("a_custom_field_2_name",
230                 "field 2 value: \u00C4-\u00D6-\u00DC".getBytes(encoding));
231         // ...and add them to the event
232         event.setField(field1);
233         event.setField(field2);
234 
235         // this is where you pass the event into the audit processing chain
236         auditEvent(event);
237 
238         // Clear the confidential information from the event
239         // (if you always want to do that, feel free to move this to the {@code auditEvent()} method.)
240         event.clear();
241 
242         // Clear the confidential information from the fields that we have created manually
243         field1.clear();
244         field2.clear();
245     }

247     /**
248      * This is an example on how to get an instance of the audit processing chain, handing off an even for processing,
249      * and handling errors that may occur during processing.
250      *
251      * @param event The event to audit
252      */
253     private void auditEvent(Event event) {
254 
255         // get an instance of Audit. You may want to use the getSinlgeton() method, or use getInstance()
256         // and manage the your own Audit singleton(s) to improve performance...
257         final Audit audit;
258         try {
259             audit = AuditFactory.getInstance(properties.getAuditClassName(), properties);
260         } catch (FactoryException e) {
261             // your error handling in case of instantiation failures goes here
262             LOG.error("Failed to retrieve an instance of the configured Audit implementation ("
263                     + properties.getAuditClassName() + ")", e);
264             return;
265         }
266 
267         // audit to the default audit stream for this demo
268         try {
269             audit.audit(event);
270         } catch (AuditException e) {
271 
272             // In case that auditing fails, you have access to an enum of error conditions.
273             // See {@link AuditErrorConditions} for a complete list.
274             switch (e.getErrorCondition()) {
275                 case CONFIGURATION:
276                     LOG.error("Audit failed because of a config error: " + AuditErrorConditions.CONFIGURATION, e);
277                     break;
278                 case UNKNOWN:
279                     LOG.error("Audit failed because of an unknown error: " + AuditErrorConditions.UNKNOWN, e);
280                     break;
281                 default:
282                     LOG.error("Auditing failed, and we don't know why. " +
283                             "Maybe new fields were added to the enum, and we have not updated our code yet", e);
284                     break;
285             }
286 
287             // typically you want to re-throw an exception here to prevent further execution of your code
288             // (assuming that you do not want to execute any operation without it being audited)
289         }
290     }
