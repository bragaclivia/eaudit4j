33  package org.beiter.michael.eaudit4j.common;
34  
35  /**
36   * The audit chain combines one or more processors into an input / output sequence of processors. When an event is
37   * submitted to an audit chain, the event is passed through all processors that are part of the audit chain, and the
38   * output of a first processor is used as the input of a second processor.
39   * <p>
40   * Implementations of this interface may produce synchronous or asynchronous audit chains.
41   * <p>
42   * Classes implementing this interface <b>must</b> be thread safe.
43   */
44  public interface Audit {
45  
46      /**
47       * Initializes the audit chain configuration
48       * <p>
49       * A class implementing this interface must provide a reasonable default configuration and handle situations where
50       * the interface's methods are called without a previous call of {@code init()} (i.e. do not throw a runtime
51       * exception).
52       * <p>
53       * A class implementing this interface must ensure that subsequent calls to this method update the class'
54       * configuration in a thread-safe way.
55       *
56       * @param properties The properties to initialize the audit chain with. Supported "additionalParameters" may
57       *                   vary with the implementing classes.
58       * @throws AuditException When the initialization fails
59       */
60      void init(CommonProperties properties)
61              throws AuditException;
62  
63      /**
64       * This method processes an event in the default audit stream.
65       * <p>
66       * The event may be modified by the processors contained in the audit stream. When the event has been processed,
67       * it is returned to the caller.
68       * <p>
69       * Note that in asynchronous implementations of this interface, the behavior is undefined. The returned event may
70       * be, for instance, the originally provided event (or a copy thereof), or simply {@code null}.
71       *
72       * @param event The event to audit
73       * @return The event after processing (for synchronous audit implementations), undefined otherwise
74       * @throws AuditException when the audit operation fails
75       */
76      Event audit(Event event)
77              throws AuditException;
78  
79      /**
80       * This method processes an event in the provided audit stream.
81       * <p>
82       * Applications may support more than one audit streams. Audit streams are event streams that are isolated from
83       * each other. As an example, an application may want to support one audit stream per tenant, plus one or more
84       * additional audit streams for global events, allowing sharing of audit data with tenants without the risk of
85       * compromising other tenants but still maintaining full integrity of the shared audit data.
86       * <p>
87       * The event may be modified by the processors contained in the audit stream. When the event has been processed,
88       * it is returned to the caller.
89       * <p>
90       * Note that in asynchronous implementations of this interface, the behavior is undefined. The returned event may
91       * be, for instance, the originally provided event (or a copy thereof), or simply {@code null}.
92       *
93       * @param event           The event to audit
94       * @param auditStreamName The audit stream to send events to
95       * @return The event after processing (for synchronous audit implementations), undefined otherwise
96       * @throws AuditException when the audit operation fails
97       */
98      Event audit(Event event, String auditStreamName)
99              throws AuditException;
100 
101     /**
102      * This method processes an event in the provided audit stream and includes a set of {@link ProcessingObjects}.
103      * <p>
104      * {@link ProcessingObjects} allow an application to inject a set of named objects into the processor. This is
105      * commonly used for objects that are managed by applications, and cannot or should not be created by an individual
106      * {@link Processor}.
107      * <p>
108      * Examples for such objects include {@link javax.sql.DataSource} objects to connect to a database, with the
109      * database connection pool being managed directly by the application.
110      * <p>
111      * The event may be modified by the processors contained in the audit stream. When the event has been processed,
112      * it is returned to the caller.
113      * <p>
114      * Note that in asynchronous implementations of this interface, the behavior is undefined. The returned event may
115      * be, for instance, the originally provided event (or a copy thereof), or simply {@code null}.
116      *
117      * @param event             The event to be processed
118      * @param auditStreamName   The audit stream to send events to
119      * @param processingObjects The processing objects available to the processors
120      * @return The event after processing (for synchronous audit implementations), undefined otherwise
121      * @throws AuditException when the audit operation fails
122      */
123     Event audit(Event event, String auditStreamName, ProcessingObjects processingObjects)
124             throws AuditException;
125 }
