33  package org.beiter.michael.eaudit4j.common;
34  
35  /**
36   * A class implementing the {@code process()} methods in the {@link Processor} interface may make modifications to an
37   * event, such as modifying existing fields, or adding new fields.
38   * <p>
39   * Possible modifications of an event that may be provided in a class implementing the {@link Processor} interface in
40   * particular include data compression and data encryption. While not all possible modification operations are
41   * reversible, some of them are.
42   * <p>
43   * Classes implementing this interface indicate that their modifications to an event are reversible, either by undoing
44   * modifications on event fields (e.g. decrypting a previously encrypted field), or by removing additional fields
45   * that were added during processing (e.g. removing a field that contains a one-way hash of the event).
46   */
47  public interface Reversible {
48  
49      /**
50       * Revert changes that have been made to an event by the {@link Processor#process(Event)} and
51       * {@link Processor#process(Event, String)} methods.
52       *
53       * @param event The event to revert changes on
54       * @return The provided event with previously made changes reverted
55       * @throws AuditException When the changes cannot be reverted
56       */
57      Event revert(Event event)
58              throws AuditException;
59  
60  }
