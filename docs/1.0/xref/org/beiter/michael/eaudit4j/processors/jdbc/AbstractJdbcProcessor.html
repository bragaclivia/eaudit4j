60  /**
61   * This processors persists events to a JDBC database. The connection method (pool, JNDI, ...) is not managed in this
62   * class and depends on the implementations extending this class.
63   * <p>
64   * The processor stores the event in a first table and, optionally if so configured, a configurable set of fields in a
65   * second table as key / value pairs. This allows searching for specific event attributes in the second table. The
66   * entries in the second table are "linked" to the first table by a shared identifier, allowing to make joins e.g. when
67   * audit events are to be analyzed or exported.
68   * <p>
69   * This implementation supports storing specific {@link Event} fields in the second table, either under their
70   * {@link Event} field name, or through an "alias", which allows storing {@link Event} fields in the second table using
71   * an alternative name ("key") to identify the field.
72   */
73  public abstract class AbstractJdbcProcessor
74          implements Processor, Reversible {

141     /**
142      * This method processes an event in the provided audit stream and includes a set of {@link ProcessingObjects}.
143      * <p>
144      * See {@link org.beiter.michael.eaudit4j.common.Audit#audit(Event, String)}.
145      *
146      * @param event             The event to audit
147      * @param auditStreamName   The audit stream to send events to
148      * @param processingObjects The processing objects available to the processors
149      * @return The event after processing (for synchronous audit implementations), undefined otherwise
150      * @throws AuditException           when the audit operation fails (e.g. the event could not be persisted)
151      * @throws NullPointerException     When the {@code auditStreamName} or {@code processingObjects} are {@code null}
152      * @throws IllegalArgumentException When {@code auditStreamName} is empty
153      */
154     @Override
155     public final Event process(final Event event,
156                                final String auditStreamName,
157                                final ProcessingObjects processingObjects)
158             throws AuditException {
159 
160         if (event == null) {
161             final String error = "The validated object 'event' is null";
162             LOG.warn(error);
163             throw new AuditException(AuditErrorConditions.INVALID_EVENT, error);
164         }
165 
166         Validate.notBlank(auditStreamName, "The validated character sequence 'auditStreamName' is null or empty");
167 
168         // this implementation does not use processing objects, but validate anyway to ensure library API compliance
169         Validate.notNull(processingObjects, "The validated object 'processingObjects' is null");
170 
171         // Make sure that the properties we need are available
172         if (properties == null) {
173 
174             final String error = "eAudit4j processor specific properties for " + this.getClass().getCanonicalName()
175                     + " have not been initialized.";
176             LOG.warn(error);
177             throw new AuditException(AuditErrorConditions.INITIALIZATION, error);
178         }
179 
180         // extract the event ID from the event, and throw an exception if the event ID is not present
181         final String eventId;
182         if (event.containsField(properties.getEventIdFieldName())) {
183             final Field eventIdField = event.getField(properties.getEventIdFieldName()); // compiler will optimize this
184             eventId = String.valueOf(eventIdField.getCharValue(properties.getStringEncoding()));
185         } else {
186             final String error = "The required field `event ID` is not present in the event. Have you configured "
187                     + "(1) a processor that adds (random) event IDs, and "
188                     + "(2) configured that processor to run before this processor in the chain, and "
189                     + "(3) configured this processor to use the correct name for the event ID field?";
190             LOG.warn(error);
191             throw new AuditException(AuditErrorConditions.CONFIGURATION, error);
192         }
193 
194         // create a key / value map of the fields that go to the index table
195         final Map<String, Field> indexedFields = getIndexedFields(event, properties);
196 
197         // serializing the event to JSON will take bit, which is why we do it outside of the transaction
198         final String eventJson = String.valueOf(event.toJson(properties.getStringEncoding()));
199 
200         // persist the event
201         persistEvent(auditStreamName, processingObjects, eventId, indexedFields, eventJson);
202 
203         // return the event unchanged
204         return event;
205     }

243     /**
244      * Create a key / value map of the indexed event fields to be added to the search table as configured
245      *
246      * @param event       The event to take the fields from
247      * @param pProperties The processor configuration
248      * @return A key / value map with the fields to add to the search table
249      * @throws AuditException When there is an invalid indexed field list configuration
250      */
251     // CHECKSTYLE:OFF
252     // this is flagged in checkstyle with a missing whitespace before '}', which is a bug in checkstyle
253     // Need to validate sizes of the indexedField split to decide whether we use a custom name, or the field name. We
254     // could do that with a null check, but that seems to be awkward.
255     // Suppress warnings about this method being too complex (can't extract a generic subroutine to reduce exec paths).
256     @SuppressWarnings({"PMD.AvoidLiteralsInIfCondition", "PMD.CyclomaticComplexity"})
257     // CHECKSTYLE:ON
258     private Map<String, Field> getIndexedFields(final Event event, final JdbcProperties pProperties)
259             throws AuditException {
260 
261         final Map<String, Field> map = new ConcurrentHashMap<>();
262 
263         // split the configured event name list, and see what we need to add to the map
264         if (pProperties.getIndexedFields() != null && !pProperties.getIndexedFields().isEmpty()) {
265 
266             // split the list of fields that we need to add to the map
267             final String[] indexedFieldNames =
268                     pProperties.getIndexedFields().split(pProperties.getIndexedFieldSeparator());
269 
270             // Go through the list of configured fields, and check if there is an alias configured.
271             // If so, use that alias - otherwise, use the field name directly.
272             // Only add the field if it exists in the event (not all events necessarily contain all fields)
273             for (final String indexedFieldName : indexedFieldNames) {
274 
275                 final String[] indexedField = indexedFieldName.split(pProperties.getIndexedFieldNameSeparator());
276                 final Field field; // the actual field
277                 final String fieldName; // the name (i.e. the key) we use for the field in the map
278 
279                 if (indexedField.length == 1) {
280 
281                     // we do not have a dedicated key configured, hence use the field name.
282                     // first, we check if the field has been set:
283                     if (event.containsField(indexedField[0])) {
284 
285                         // the field exists for this event, which means we can pull it
286                         field = event.getField(indexedField[0]);
287                         fieldName = field.getName();
288                     } else {
289 
290                         // the field does not exist in this event, hence proceed to the next field
291                         continue;
292                     }
293                 } else if (indexedField.length == 2) {
294 
295                     // we do have a dedicated key configured
296                     // first, we check if the field has been set:
297                     if (event.containsField(indexedField[0])) {
298 
299                         // the field exists for this event, which means we can pull it
300                         field = event.getField(indexedField[0]);
301                         fieldName = indexedField[1];
302                     } else {
303 
304                         // the field does not exist in this event, hence proceed to the next field
305                         continue;
306                     }
307                 } else {
308                     // We have less than 1 and more than 2 field name components.
309                     // This should never happen.
310                     final String error = "The Event field name / index mapping is invalid. ";
311                     LOG.warn(error);
312                     throw new AuditException(AuditErrorConditions.CONFIGURATION, error);
313                 }
314 
315                 // add the field to the map
316                 map.put(fieldName, field);
317             }
318         }
319 
320         // return the map
321         return map;
322     }

324     /**
325      * Persist an event to the database
326      *
327      * @param auditStreamName   The name of the audit stream
328      * @param processingObjects The provided processing objects
329      * @param eventId           The unique event ID of the event
330      * @param indexedFields     The fields that should go in the indexed fields table
331      * @param eventJson         The serialized event
332      * @throws AuditException When The database operation fails
333      */
334     // CHECKSTYLE:OFF
335     // this is flagged in checkstyle with a missing whitespace before '}', which is a bug in checkstyle
336     // suppress warnings about the DB connection not being closed - seems as if that checker does not get it
337     // suppress warnings about comparing to "1" - changing this seems to be over-engineering...
338     // suppress warnings about nested catches and re-throwing of exceptions, with the cause being lost - can't avoid.
339     // suppress warnings about this method being too long (not much point in splitting up this one!)
340     // suppress warnings about this method being too complex (can't extract a generic subroutine to reduce exec paths)
341     @SuppressWarnings({"PMD.CloseResource", "PMD.AvoidLiteralsInIfCondition", "PMD.PreserveStackTrace", "PMD.ExcessiveMethodLength", "PMD.NPathComplexity", "PMD.CyclomaticComplexity", "PMD.StdCyclomaticComplexity", "PMD.ModifiedCyclomaticComplexity"})
342     // The SQL statement is retrieved from the configuration, and the admin is trusted
343     // Same open data source comment as above - may be a problem with the checker and the code complexity
344     // The OBL_UNSATISFIED_OBLIGATION checker (which is marked experimental) does not seem to work properly
345     @edu.umd.cs.findbugs.annotations.SuppressWarnings({"SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING", "ODR_OPEN_DATABASE_RESOURCE", "OBL_UNSATISFIED_OBLIGATION"})
346     // CHECKSTYLE:ON
347     private void persistEvent(final String auditStreamName, final ProcessingObjects processingObjects,
348                               final String eventId, final Map<String, Field> indexedFields,
349                               final String eventJson)
350             throws AuditException {
351 
352         // get a database connection
353         final Connection con;
354         try {
355             con = getConnection(properties, processingObjects);
356         } catch (FactoryException e) {
357             final String error = "Cannot retrieve database connection";
358             LOG.warn(error, e);
359             throw new AuditException(AuditErrorConditions.PROCESSING, error, e);
360         }
361 
362         // get a reasonable fallback for the auto-commit
363         boolean autoCommit = MapBasedConnPropsBuilder.build(properties.getAdditionalProperties()).isDefaultAutoCommit();
364 
365         // execute the DB transaction
366         PreparedStatement psEvent = null;
367         PreparedStatement psIndex = null;
368         try {
369 
370             // we will commit manually
371             autoCommit = con.getAutoCommit();
372             con.setAutoCommit(false);
373 
374             // create prepared statement for the event, and populate it
375             psEvent = con.prepareStatement(properties.getInsertEventSqlStmt());
376             psEvent.setString(1, eventId);
377             psEvent.setString(2, auditStreamName);
378             psEvent.setString(3, eventJson);
379             final int eventRows = psEvent.executeUpdate();
380 
381             // check if the operation has been executed properly:
382             if (eventRows != 1) {
383                 final String error = "Error when persisting the audit event. The operation should have affected '1' "
384                         + "index row per field, but a total of '" + eventRows + "' rows was affected";
385                 LOG.warn(error);
386                 throw new AuditException(AuditErrorConditions.PROCESSING, error);
387             }
388 
389             // only add indexed fields if there are indexed fields configured... obviously.
390             if (properties.getIndexedFields() != null && !properties.getIndexedFields().isEmpty()) {
391 
392                 // create a prepared statement for the index, and batch the operations
393                 psIndex = con.prepareStatement(properties.getInsertIndexedFieldSqlStmt());
394                 for (final Map.Entry<String, Field> entry : indexedFields.entrySet()) {
395 
396                     final String fieldName = entry.getKey();
397                     String fieldValue = String.valueOf(entry.getValue().getCharValue(properties.getStringEncoding()));
398 
399                     // truncate the field if needed
400                     if (fieldValue.length() > properties.getIndexedFieldsMaxLength()) {
401                         fieldValue = fieldValue.substring(0, properties.getIndexedFieldsMaxLength());
402                     }
403 
404                     // convert the field value to lower if needed
405                     if (properties.isIndexedFieldsToLower()) {
406                         // We use the platform's default locale here. This could be made configurable, but we wait
407                         // until someone asks for it. Using an existing object here will help a lot with performance,
408                         // and the alternative of using a constant (e.g. Locale.US) is not very appealing either.
409                         // This is a pattern used throughout the library. Search for Locale.getDefault() to find all
410                         // locations.
411                         fieldValue = fieldValue.toLowerCase(Locale.getDefault());
412                     }
413 
414                     // normalize the searchable String value to NFC form for deterministic search
415                     fieldValue = Normalizer.normalize(fieldValue, Normalizer.Form.NFC);
416 
417                     psIndex.setString(1, eventId);
418                     psIndex.setString(2, auditStreamName);
419                     psIndex.setString(3, fieldName);
420                     psIndex.setString(4, fieldValue);
421 
422                     psIndex.addBatch();
423                 }
424                 final int[] indexRows = psIndex.executeBatch();
425 
426                 // check if the operation has been executed properly:
427                 // We do not know the number of field operations for sure (this varies with the number of
428                 // fields-to-be-persisted present in the actual event), but we can check the execution status of each
429                 // field. Not much use without knowing which field, but it is better than nothing and we probably do
430                 // not want to over-engineer the error processing here.
431                 for (final int indexRow : indexRows) {
432                     if (indexRow != 1) {
433                         final String error = "Error when persisting the audit event. The operation should have "
434                                 + "affected '1' index row per field, but a total of '" + indexRow
435                                 + "' rows was affected";
436                         LOG.warn(error);
437                         throw new AuditException(AuditErrorConditions.PROCESSING, error);
438                     }
439                 }
440             }
441 
442             // commit
443             con.commit();
444 
445             // Reset the auto-commit and close the resources.
446             // We add this code twice, here and in the finalizer. This is because we never know when the finalizer
447             // will actually be executed, and closing the connection here immediately in the non-error case prevents
448             // resource leaks. Repeating the same code below in the finalizer is just a safety net.
449             con.setAutoCommit(autoCommit);
450             if (!psEvent.isClosed()) {
451                 psEvent.close();
452             }
453             if (psIndex != null && !psIndex.isClosed()) {
454                 psIndex.close();
455             }
456             if (!con.isClosed()) {
457                 con.close();
458             }
459         } catch (SQLException e) {
460 
461             final String error = "Unrecoverable error when executing the SQL transaction";
462             LOG.warn(error, e);
463 
464             try {
465 
466                 // try to roll back
467                 con.rollback();
468 
469             } catch (SQLException e1) {
470 
471                 final String error1 = "Unrecoverable error when executing the SQL transaction. "
472                         + "Also encountered an unrecoverable error when trying to roll back the SQL transaction. "
473                         + "The transaction has not been executed, has not been closed, and has not been rolled back!";
474                 LOG.warn(error, e1);
475                 throw new AuditException(AuditErrorConditions.PROCESSING, error1, e1);
476             }
477 
478             throw new AuditException(AuditErrorConditions.PROCESSING, error, e);
479 
480         } finally {
481 
482             // reset the auto-commit and close the resources only if they have not already been closed in the happy path
483             try {
484                 if (!con.isClosed() && con.getAutoCommit() != autoCommit) {
485                     con.setAutoCommit(autoCommit);
486                 }
487                 if (psEvent != null && !psEvent.isClosed()) {
488                     psEvent.close();
489                 }
490                 if (psIndex != null && !psIndex.isClosed()) {
491                     psIndex.close();
492                 }
493                 if (!con.isClosed()) {
494                     con.close();
495                 }
496             } catch (SQLException e) {
497                 final String error = "Cannot close the database connection";
498                 LOG.warn(error, e);
499             }
500         }
501     }
